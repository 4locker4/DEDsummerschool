83void SquareSolver (struct Coefficient parameters, struct Roots* decision) // Returns the number of roots
84{
85 assert (isfinite (parameters.a));
86 assert (isfinite (parameters.b));
87 assert (isfinite (parameters.c));
88
89 assert (decision != NULL);
90
91 if (IsZero (parameters.c)) // If c == 0
92 {
93 if (IsZero (parameters.b))
94 {
95 decision->x1 = decision->x2 = 0;
96 decision->nRoots = ONE_ROOT;
97 return;
98 }
99 decision->x1 = 0;
100 decision->x2 = (parameters.b / -parameters.a);
101
102 decision->nRoots = TWO_ROOTS;
103
104 return;
105 }
106
107 double discriminant = parameters.b * parameters.b - 4 * parameters.a * parameters.c; // Calculate the discriminant
108 if (discriminant <= -EPSILON) // Testing for non-negative discriminant
109 {
110 decision->nRoots = ZERO_ROOTS;
111 return;
112 }
113
114 double sqrtDiscriminant = sqrt (discriminant); // Root of the discriminant
115 if (IsZero (sqrtDiscriminant)) // One root
116 {
117 decision->x1 = decision->x2 = (-parameters.b / parameters.a);
118 decision->nRoots = ONE_ROOT;
119 return;
120 }
121 else // 2 roots, sqrt_discriminant > EPSILAN
122 {
123 decision->x1 = (-parameters.b + sqrtDiscriminant) / (2 * parameters.a); // First root
124 decision->x2 = (-parameters.b - sqrtDiscriminant) / (2 * parameters.a); // Second root
125 decision->nRoots = TWO_ROOTS;
126 return;
127 }
