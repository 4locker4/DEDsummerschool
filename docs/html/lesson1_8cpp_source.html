111int SquareSolver (double a, double b, double c, double* x1, double* x2) // Returns the number of roots
112{
113 assert(isfinite (a));
114 assert(isfinite (b));
115 assert(isfinite (c));
116
117 assert(x1 != x2);
118 assert(x1 != NULL);
119 assert(x2 != NULL);
120
121 double discriminant = b * b - 4 * a * c; // Calculate the discriminant
122
123 if (discriminant <= 0) // Testing for non-negative discriminant
124 return ZERO_ROOTS;
125
126 double sqrtDiscriminant = sqrt(discriminant); // Root of the discriminant
127
128 if (IsZero(sqrtDiscriminant)) // One root
129 {
130 *x1 = *x2 = -b / (2 * a);
131 return ONE_ROOT;
132 }
133 else // 2 roots, sqrt_discriminant > EPSILAN
134 {
135 *x1 = (-b + sqrtDiscriminant) / (2 * a); // First root
136 *x2 = (-b - sqrtDiscriminant) / (2 * a); // Second root
137 return TWO_ROOTS;
138 }
